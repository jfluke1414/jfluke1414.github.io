SELECT 
    SEGMENT_TYPE
    , SEGMENT_NAME
    , TABLESPACE_NAME
    , SUM(BYTES)
FROM DBA_SEGMENTS    
WHERE SEGMENT_TYPE IN ('TABLE', 'TABLE PARTITION', 'TABLE SUBPARTITION')
AND UPPER(OWNER) = 'DY15'
GROUP BY SEGMENT_TYPE 
    , SEGMENT_NAME
    , TABLESPACE_NAME
